1. Architecture Overview
This guide covers exactly how to upgrade SubtitleAI Pro's video editor from a basic caption tool into a premium, smooth, multi-agent editing experience. Each section maps directly to your existing codebase in VideoEditorPage.tsx, groq_service.py, and export.py.

Why smoothness feels broken right now
Your current stack renders caption overlays via a <video> element with CSS absolutely-positioned text. Every caption update causes a React re-render which repaints the DOM. There is no compositor, no frame timing, and no GPU acceleration. This is why it lags.

1.2 Data Flow

Key Insight: Proxy Editing
Never edit the original 4K file in the browser. Convert to 360p proxy on upload. The editor works on the proxy. On export, the backend re-applies all edits to the original at full quality. This is exactly what Premiere Pro and DaVinci Resolve do.

2. Rendering Engine (WebGL)
This is the most important change. Replace all CSS-based caption rendering with a WebGL compositor that runs on the GPU. Every frame is composited in the browser at 60fps without touching the DOM.

2.1 File to Create: src/engine/Compositor.ts
Create this new file. It owns the entire rendering pipeline — video decode ? frame extraction ? WebGL composite ? canvas display.
// src/engine/Compositor.ts
export class Compositor {
  private canvas: HTMLCanvasElement;
  private gl: WebGL2RenderingContext;
  private videoElement: HTMLVideoElement;
  private textureVideo: WebGLTexture | null = null;
  private program: WebGLProgram | null = null;
  private rafId: number = 0;

  constructor(canvas: HTMLCanvasElement, video: HTMLVideoElement) {
    this.canvas = canvas;
    this.video = video;
    this.gl = canvas.getContext('webgl2')!;
    this.initShaders();
    this.initTexture();
  }

  private initShaders() {
    const vertSrc = `
      attribute vec2 a_pos;
      attribute vec2 a_uv;
      varying vec2 v_uv;
      void main() {
        gl_Position = vec4(a_pos, 0.0, 1.0);
        v_uv = a_uv;
      }
    `;
    const fragSrc = `
      precision mediump float;
      uniform sampler2D u_video;
      varying vec2 v_uv;
      void main() {
        gl_FragColor = texture2D(u_video, v_uv);
      }
    `;
    this.program = this.buildProgram(vertSrc, fragSrc);
  }

  render(cues: ActiveCue[], currentTimeMs: number) {
    const gl = this.gl;
    gl.clear(gl.COLOR_BUFFER_BIT);
    this.drawVideoFrame();       // GPU: upload video texture
    this.drawCaptions(cues);     // GPU: overlay text via Canvas2D ? texture
    this.drawBrollOverlay();     // GPU: blend B-roll clip if active
  }
}

Why WebGL instead of CSS?
CSS renders on the main thread. WebGL renders on the GPU compositor thread. With CSS you get 15-30fps during editing. With WebGL you get stable 60fps. The difference is immediately visible to users.
2.2 Caption Rendering: Canvas2D ? WebGL Texture
Captions are drawn on an offscreen Canvas2D, then uploaded as a WebGL texture and blended over the video frame. This means any font, any style, zero DOM cost.
// src/engine/CaptionRenderer.ts
export class CaptionRenderer {
  private offscreen: OffscreenCanvas;
  private ctx: OffscreenCanvasRenderingContext2D;

  constructor(width: number, height: number) {
    this.offscreen = new OffscreenCanvas(width, height);
    this.ctx = this.offscreen.getContext('2d')!;
  }

  renderCue(cue: SubtitleCue, style: CaptionStyle): ImageBitmap {
    const ctx = this.ctx;
    ctx.clearRect(0, 0, this.offscreen.width, this.offscreen.height);
    ctx.font = `${style.fontWeight} ${style.fontSize}px ${style.fontFamily}`;
    ctx.textAlign = 'center';
    ctx.strokeStyle = style.strokeColor;
    ctx.lineWidth = style.strokeWidth;
    ctx.strokeText(cue.text, this.offscreen.width / 2, this.getY(style));
    ctx.fillStyle = style.color;
    ctx.fillText(cue.text, this.offscreen.width / 2, this.getY(style));
    return this.offscreen.transferToImageBitmap();
  }
}


3. Timeline + State Engine (Zustand)
Your current state is scattered across useState hooks in VideoEditorPage. Centralise everything into a Zustand store with computed selectors. This is the state layer your Compositor will read from on every frame.

3.1 Create: src/store/timelineStore.ts
// src/store/timelineStore.ts
import { create } from 'zustand';
import { subscribeWithSelector } from 'zustand/middleware';

interface TimelineState {
  // Playback
  currentTimeMs: number;
  durationMs: number;
  isPlaying: boolean;
  playbackRate: number;

  // Tracks
  subtitleTrack: SubtitleCue[];
  brollTrack: BrollClip[];
  musicTrack: MusicClip | null;
  gapCuts: GapCut[];

  // Selection
  selectedCueId: string | null;
  zoom: number;          // px per second
  scrollOffsetMs: number;

  // Actions
  setTime: (ms: number) => void;
  setPlaying: (v: boolean) => void;
  updateCue: (id: string, patch: Partial<SubtitleCue>) => void;
  addBroll: (clip: BrollClip) => void;
  setZoom: (z: number) => void;
}

export const useTimeline = create<TimelineState>()(
  subscribeWithSelector((set, get) => ({
    currentTimeMs: 0,
    durationMs: 0,
    isPlaying: false,
    playbackRate: 1,
    subtitleTrack: [],
    brollTrack: [],
    musicTrack: null,
    gapCuts: [],
    selectedCueId: null,
    zoom: 100,
    scrollOffsetMs: 0,

    setTime: (ms) => set({ currentTimeMs: ms }),
    setPlaying: (v) => set({ isPlaying: v }),
    updateCue: (id, patch) => set(s => ({
      subtitleTrack: s.subtitleTrack.map(c => c._id === id ? { ...c, ...patch } : c)
    })),
    addBroll: (clip) => set(s => ({ brollTrack: [...s.brollTrack, clip] })),
    setZoom: (z) => set({ zoom: Math.max(20, Math.min(500, z)) }),
  }))
);

4. Web Workers + OffscreenCanvas
Move all heavy computation off the main thread. The main thread only handles user events and DOM updates. Everything else — frame analysis, waveform generation, thumbnail extraction — runs in a worker.

4.1 Create: src/workers/renderWorker.ts
// src/workers/renderWorker.ts  (runs in dedicated worker thread)

let offscreenCanvas: OffscreenCanvas | null = null;
let gl: WebGL2RenderingContext | null = null;

self.onmessage = (e: MessageEvent) => {
  const { type, payload } = e.data;

  switch (type) {
    case 'INIT':
      offscreenCanvas = payload.canvas;  // transferred, not cloned
      gl = offscreenCanvas.getContext('webgl2')!;
      self.postMessage({ type: 'READY' });
      break;

    case 'RENDER_FRAME':
      // payload: { bitmap: ImageBitmap, cues: ActiveCue[], style: CaptionStyle }
      renderFrame(payload.bitmap, payload.cues, payload.style);
      self.postMessage({ type: 'FRAME_DONE' });
      break;

    case 'GENERATE_WAVEFORM':
      const peaks = generateWaveformPeaks(payload.audioBuffer, payload.width);
      self.postMessage({ type: 'WAVEFORM_DONE', peaks }, [peaks.buffer]);
      break;
  }
};

4.2 Main Thread: Spawn and Transfer Canvas
// src/engine/WorkerBridge.ts
export class WorkerBridge {
  private worker: Worker;

  constructor(canvas: HTMLCanvasElement) {
    this.worker = new Worker(
      new URL('../workers/renderWorker.ts', import.meta.url),
      { type: 'module' }
    );
    // Transfer canvas ownership to worker — main thread can't touch it after this
    const offscreen = canvas.transferControlToOffscreen();
    this.worker.postMessage({ type: 'INIT', payload: { canvas: offscreen } },
      [offscreen]  // transferable — zero-copy
    );
  }

  renderFrame(bitmap: ImageBitmap, cues: SubtitleCue[], style: CaptionStyle) {
    this.worker.postMessage(
      { type: 'RENDER_FRAME', payload: { bitmap, cues, style } },
      [bitmap]  // transfer bitmap to avoid clone cost
    );
  }
}

Transferable Objects (Zero-Copy)
When you post ImageBitmap or ArrayBuffer, add them to the transfer list [bitmap]. The browser transfers ownership without copying the data. A 1920x1080 frame is 8MB — copying it 60x/sec = 480MB/s of unnecessary memory pressure. Transfer = 0 bytes copied.

5. Video Playback Architecture
The <video> element handles decoding (hardware accelerated). Your WebGL compositor handles the display. They communicate via requestVideoFrameCallback — the browser's API for perfect frame-timing sync.

5.1 Create: src/engine/PlaybackController.ts
// src/engine/PlaybackController.ts
export class PlaybackController {
  private video: HTMLVideoElement;
  private compositor: Compositor;
  private setTime: (ms: number) => void;
  private rvfcHandle: number = 0;

  constructor(video: HTMLVideoElement, compositor: Compositor,
              setTime: (ms: number) => void) {
    this.video = video;
    this.compositor = compositor;
    this.setTime = setTime;
  }

  start() {
    // requestVideoFrameCallback fires EXACTLY when a video frame is ready
    // Much more accurate than requestAnimationFrame for video sync
    this.rvfcHandle = this.video.requestVideoFrameCallback(
      this.onFrame.bind(this)
    );
  }

  private onFrame(now: DOMHighResTimeStamp, meta: VideoFrameCallbackMetadata) {
    const ms = meta.mediaTime * 1000;
    this.setTime(ms);           // update Zustand — triggers selector recalc
    const activeCues = getActiveCues(ms);
    const bitmap = this.video.requestVideoFrameAsImageBitmap?.() ??
                   grabFrame(this.video);
    this.compositor.render(bitmap, activeCues, ms);  // GPU render
    // Re-register for next frame
    if (!this.video.paused) {
      this.rvfcHandle = this.video.requestVideoFrameCallback(
        this.onFrame.bind(this)
      );
    }
  }
}

5.2 Proxy Video System
This is the most impactful performance trick. Never edit in the browser using the original file.

# backend/app/services/proxy_service.py
import asyncio, subprocess, os

async def generate_proxy(input_path: str, project_id: str) -> str:
    proxy_path = f'uploads/proxy_{project_id}_360p.mp4'
    cmd = [
        'ffmpeg', '-y', '-i', input_path,
        '-vf', 'scale=640:360:force_original_aspect_ratio=decrease',
        '-c:v', 'libx264', '-preset', 'ultrafast', '-crf', '32',
        '-c:a', 'aac', '-b:a', '64k',
        '-movflags', '+faststart',   # ? makes streaming start instantly
        proxy_path
    ]
    proc = await asyncio.create_subprocess_exec(
        *cmd, stdout=asyncio.subprocess.DEVNULL,
        stderr=asyncio.subprocess.DEVNULL
    )
    await proc.wait()
    return proxy_path

6.1 Frame Skipping During Timeline Scrub
While the user drags the timeline scrubber, skip rendering every frame. Only render when the drag stops or every 4th frame. Eliminates the lag that makes timeline scrubbing feel sticky.

6.5 Chunk Upload (Critical for Large Files)
Replace the current single-request upload with chunked multipart upload. A 2GB file on a mobile connection needs to survive interruptions and resume.
// src/lib/chunkedUpload.ts
const CHUNK_SIZE = 5 * 1024 * 1024;  // 5MB chunks

export async function chunkedUpload(
  file: File,
  projectId: string,
  onProgress: (pct: number) => void
): Promise<string> {
  const totalChunks = Math.ceil(file.size / CHUNK_SIZE);
  let uploadedBytes = 0;

  for (let i = 0; i < totalChunks; i++) {
    const chunk = file.slice(i * CHUNK_SIZE, (i + 1) * CHUNK_SIZE);
    const form = new FormData();
    form.append('chunk', chunk);
    form.append('chunk_index', String(i));
    form.append('total_chunks', String(totalChunks));
    form.append('project_id', projectId);

    await api.post('/v1/media/chunk', form);
    uploadedBytes += chunk.size;
    onProgress(uploadedBytes / file.size);
  }
  // Tell backend to assemble chunks
  const { data } = await api.post('/v1/media/assemble', { project_id: projectId });
  return data.file_path;
}

async def stream_transcription_progress(
    project_id: str,
    current_user: dict = Depends(get_current_active_user),
    db = Depends(get_database),
):
    async def event_generator():
        while True:
            job = await db.transcription_jobs.find_one({
                'project_id': project_id
            })
            if not job: break
            status = job.get('status', 'unknown')
            progress = job.get('progress_pct', 0)
            yield f'data: {json.dumps({"status": status, "progress": progress})}\n\n'
            if status in ('complete', 'error'): break
            await asyncio.sleep(1)  # poll every 1s

    return StreamingResponse(
        event_generator(),
        media_type='text/event-stream',
        headers={'Cache-Control': 'no-cache', 'X-Accel-Buffering': 'no'}
    )

export const useTranscriptionStream = (projectId: string) => {
  const [progress, setProgress] = useState(0);
  const [status, setStatus] = useState<'idle'|'queued'|'processing'|'complete'|'error'>('idle');

    es.onmessage = (e) => {
      const { status: s, progress: p } = JSON.parse(e.data);
      setStatus(s);
      setProgress(p);
      if (s === 'complete' || s === 'error') es.close();
    };
    return () => es.close();
  }, [projectId]);

  return { progress, status };
};

async def run_agents(
    project_id: str,
    agents: List[str],  # ['broll', 'music', 'gaps', 'color']
    background_tasks: BackgroundTasks,
    current_user: dict = Depends(get_current_active_user),
    db = Depends(get_database),
):
    job = {
        'project_id': project_id,
        'agents': agents,
        'status': 'running',
        'agent_status': {a: 'queued' for a in agents},
        'created_at': datetime.now(timezone.utc)

    background_tasks.add_task(
        run_parallel_agents, str(result.inserted_id), project_id, agents, db

async def run_parallel_agents(job_id, project_id, agents, db):
    # Run all agents concurrently with asyncio.gather
    tasks = []
    if 'broll' in agents:
        tasks.append(run_broll_agent(job_id, project_id, db))
    if 'music' in agents:
        tasks.append(run_music_agent(job_id, project_id, db))
    if 'gaps' in agents:
        tasks.append(run_gap_agent(job_id, project_id, db))
    await asyncio.gather(*tasks, return_exceptions=True)

8.2 Gap Cutting Agent (New Service)
# backend/app/services/gap_service.py

async def detect_silence_gaps(
    file_path: str,
    noise_db: float = -35,
    min_gap_ms: int = 500
) -> list[dict]:
    cmd = [
        'ffmpeg', '-i', file_path,
        '-af', f'silencedetect=noise={noise_db}dB:d={min_gap_ms/1000}',
        '-f', 'null', '-'
    ]
    proc = await asyncio.create_subprocess_exec(
        *cmd, stdout=asyncio.subprocess.DEVNULL,
        stderr=asyncio.subprocess.PIPE
    )
    _, stderr = await proc.communicate()
    return parse_silence_output(stderr.decode())

def parse_silence_output(stderr: str) -> list[dict]:
    gaps = []
    starts = re.findall(r'silence_start: ([d.]+)', stderr)
    ends = re.findall(r'silence_end: ([d.]+)', stderr)
    for s, e in zip(starts, ends):
        gaps.append({
            'start_ms': int(float(s) * 1000),
            'end_ms': int(float(e) * 1000),
            'type': 'silence'
        })
    return gaps
